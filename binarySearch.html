Iterative

#include stdio.h
#include stdbool.h
int main() {
bool Found = false;
int low, high, mid, key, n, i;
int array[20];
printf("Enter number of elements in the array: ");
scanf("%d", &n);
printf("Enter elements: ");
for (i = 0; i < n; i++) {
scanf("%d", &array[i]);
}
printf("Enter key to search: ");
scanf("%d", &key);
low = 0;
high = n - 1;
while (low <= high) {
mid = (low + high) / 2;
if (array[mid] == key) {
printf("Found at index %d ", mid);
Found = true;
break;
} else if (key > array[mid]) {
low = mid + 1;
} else {
high = mid - 1;
}
}
if (!Found)
printf("Not found in the array");
return 0;
}

// Binary Search function
int binsearch(int a[], int x, int low, int high) {
int mid;
if (low > high)
return -1;
mid = (low + high) / 2;
if (x == a[mid]) {
return (mid);
} else if (x < a[mid]) {
binsearch(a, x, low, mid - 1);
} else { binsearch(a, x, mid + 1, high);
}
