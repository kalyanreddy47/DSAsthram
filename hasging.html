Implementation of Hashing

#include stdio.h
#include stdlib.h
#define TABLE_SIZE 20
#define DELETED -1
int ht[TABLE_SIZE];
int hash_function(int ele) {
return ele % TABLE_SIZE;
}
void insert(int element) {
int index = hash_function(element);
int n = 0;
while (ht[index] != 0 && ht[index] != DELETED && n++ < TABLE_SIZE) {
index = hash_function(index + 1);
}
if (n == TABLE_SIZE) {
printf("Hash Table Full\n");
} else {
ht[index] = element;
}
}
void delete_element(int element) {
int index = hash_function(element);
int n = 0;
while (n++ < TABLE_SIZE) {
if (ht[index] == 0) {
printf("Not Found\n");
return;
}
if (ht[index] == element) {
ht[index] = DELETED;
printf("Deleted\n");
return;
}
index = hash_function(index + 1);
}
printf("Not Found\n");
}
void display() {
for (int i = 0; i < TABLE_SIZE; i++) {
printf("%d ", ht[i]);
}
printf("\n");
}
void search(int element) {
int index = hash_function(element);
int n = 0;
while (n++ < TABLE_SIZE) {
if (ht[index] == 0) {
printf("Not Found\n");
return;
}
if (ht[index] == element) {
printf("Element Found in Hash Table\n");
return;
}
index = hash_function(index + 1);
}
printf("Not Found\n");
}
int main() {
int choice, element;
for (int i = 0; i < TABLE_SIZE; i++) {
ht[i] = 0;
}
while (1) {
printf("1-Insert item in the Hash Table\n2-Remove item from the Hash Table\n3-Display
a Hash Table\n4-Search item in the Hash Table\n5-Exit\n");
printf("Enter your choice: ");
scanf("%d", &choice);
switch (choice) {
case 1:
printf("Enter Element to Insert: ");
scanf("%d", &element);
insert(element);
break;
case 2:
printf("Enter Element to Delete: ");
scanf("%d", &element);
delete_element(element);
break;
case 3:
display();
break;
case 4:
printf("Enter Element to Search for: ");
scanf("%d", &element);
search(element);
break;
case 5:
printf("Exiting...\n");
return 0;
default:
printf("Enter a valid option\n");
break;
}
}
return 0;
}
